	message( "BUILDING WITH OPENCL BACKEND")

	message( "OPENCL NOT FOUND")
	set(ENABLE_OPENCL 0)
	add_definitions( -DENABLE_OPENCL=${ENABLE_OPENCL} )

	target_link_libraries(vx_rpp amd_rpp boost_filesystem  boost_system)
